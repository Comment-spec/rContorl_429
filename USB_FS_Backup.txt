/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* Configure the system clock */
  SystemClock_Config();

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_SPI1_Init();
  MX_DMA_Init();
  MX_USART3_UART_Init();
  MX_RNG_Init();
  MX_TIM1_Init();
  MX_USB_DEVICE_Init();

  /* DWM3000 configuration--------------------------------------------------------*/
  memset(rx_buffer,0,sizeof(rx_buffer));
  memset(tx_buffer,0,sizeof(tx_buffer));
  memset(test_buffer,0,sizeof(test_buffer));

  /* Configure SPI rate, DW3000 supports up to 38 MHz */
  //port_set_dw_ic_spi_fastrate();

  /* Reset DW IC */
  reset_DWIC(); /* Target specific drive of RSTn line into DW IC low for a period. */
  Sleep(2); // Time needed for DW3000 to start up (transition from INIT_RC to IDLE_RC, or could wait for SPIRDY event)
  while (!dwt_checkidlerc()) /* Need to make sure DW IC is in IDLE_RC before proceeding */
  	  { };
  if (dwt_initialise(DWT_DW_INIT) == DWT_ERROR)
  {
	  while (1)
	  { };
  }

  /* Enabling LEDs here for debug so that for each TX the D1 LED will flash on DW3000 red eval-shield boards. */
  dwt_setleds(DWT_LEDS_ENABLE | DWT_LEDS_INIT_BLINK) ;
  /* Configure DW IC. See NOTE 5 below. */
  if(dwt_configure(&config)) /* if the dwt_configure returns DWT_ERROR either the PLL or RX calibration has failed the host should reset the device */
  {
	  while (1)
	  { };
  }
  /* Configure the TX spectrum parameters (power PG delay and PG Count) */
  dwt_configuretxrf(&txconfig_options);

  /* Create the thread(s) */
  /* definition and creation of dwTransmission */
  osThreadDef(dwTransmission, dwTx, osPriorityNormal, 0, 128);
  dwTransmissionHandle = osThreadCreate(osThread(dwTransmission), NULL);

  /* definition and creation of dwReceive */
  osThreadDef(dwReceive, dwRx, osPriorityRealtime, 0, 128);
  dwReceiveHandle = osThreadCreate(osThread(dwReceive), NULL);

  /*Turn RX on*/
  dwt_rxenable(DWT_START_RX_IMMEDIATE);
  /* Start scheduler */
  osKernelStart();

  /* We should never get here as control is now taken by the scheduler */
  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {  }
}

/**
  * @brief  Function implementing the dwTransmission thread.
  * @param  argument: Not used
  * @retval None
  */
/* USER CODE END Header_dwTx */
void dwTx(void const * argument)
{
  /* Infinite loop */
  for(;;)
  {
	  HAL_GPIO_TogglePin(GPIOB, GPIO_PIN_7);
	  if (memcmp(tx_buffer, test_buffer, FRAME_LEN_MAX) != 0) {
		  dwt_writefastCMD(CMD_TXRXOFF);
		  dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_ALL_RX_ERR);
		  dwt_setdwstate(DWT_DW_IDLE_RC);

		  /* Write frame data to DW IC and prepare transmission. See NOTE 3 below\n\r.*/
		  dwt_writetxdata(FRAME_LENGTH-FCS_LEN, tx_buffer, 0); /* Zero offset in TX buffer. */
		  dwt_writetxfctrl(FRAME_LENGTH, 0, 0); /* Zero offset in TX buffer, no ranging. */

		  /* Start transmission. */
		  dwt_starttx(DWT_START_TX_IMMEDIATE);

		  /* Poll DW IC until TX frame sent event set.*/
		  while (!(dwt_read32bitreg(SYS_STATUS_ID) & SYS_STATUS_TXFRS_BIT_MASK))
		  { };

		  /* Clear TX frame sent event. */
		  dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_TXFRS_BIT_MASK);

		  //dwt_writefastCMD(CMD_TXRXOFF);
		  dwt_rxenable(DWT_START_RX_IMMEDIATE);

		  memset(tx_buffer, 0, sizeof(tx_buffer));
	  }

    osDelay(0.01);
  }

}

/* USER CODE BEGIN Header_dwRx */
/**
* @brief Function implementing the dwReceive thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_dwRx */
void dwRx(void const * argument)
{
  /* Infinite loop */
  for(;;)
  {
	  status_reg = dwt_read32bitreg(SYS_STATUS_ID) & (SYS_STATUS_RXFCG_BIT_MASK | SYS_STATUS_ALL_RX_ERR);
	  frame_len = dwt_read32bitreg(RX_FINFO_ID) & RX_FINFO_RXFLEN_BIT_MASK;

	  if (status_reg & SYS_STATUS_RXFCG_BIT_MASK) {
		  /* A frame has been received, copy it to our local buffer. */
		  memset(rx_buffer, '\0', sizeof(rx_buffer));
		  if ((dwt_read32bitreg(RX_FINFO_ID) & RX_FINFO_RXFLEN_BIT_MASK) <= FRAME_LEN_MAX) {
			  dwt_readrxdata(rx_buffer, (frame_len-FCS_LEN), 0); /* No need to read the FCS/CRC. */
		  }

		  /* Clear good RX frame event in the DW IC status register. */
		  dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_RXFCG_BIT_MASK);

		  dwt_rxenable(DWT_START_RX_IMMEDIATE);

		  CDC_Transmit_FS(rx_buffer, (frame_len-FCS_LEN));
		  //osDelay(10);
	  }
      else
      {
          /* Clear RX error events in the DW IC status register. */
          dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_ALL_RX_ERR);
      }
    osDelay(0.01);
  }
  /* USER CODE END dwRx */
}
